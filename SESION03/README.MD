class Fraccion {
    private int num;
    private int den;

    Fraccion() {
        // System.out.println("Nuevo objeto Fraccion creado!");
    }
    Fraccion(int num, int den) {
        this.num = num;
        this.den = den;
    }
    void set(int num, int den) {
        this.num = num;
        this.den = den;
    }
    void setNum(int x) {
        num = x;
    }
    void setDen(int x) {
        if (x == 0)
            den = 1;
        else
            den = x;
    }
    int getNum() {
        return num;
    }
    int getDen() {
        return den;
    }
}
